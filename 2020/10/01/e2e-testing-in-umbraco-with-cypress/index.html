Useful
Commands


                        
// Run code before all tests
                        before(() => {
                            ...
                        })

                        
// Run code before each test
                        beforeEach(() => {
                            ... 
                        })

                    

                        
// Navigate to a page
                        cy.visit('https://example.cypress.io')

                        
// Get a dom element, and click it
                        cy.get('.btn--submit').click()

                        
// Get a the li element that
                        // contains the text 'Hello World'
                        cy.get('li').contains('Hello World')

                    

                        
// Type into an input
                        cy.get('.name').type('John Smith')

                        
// Select an select list option
                        cy.get('.country').select('United Kingdom')

                        
// Submit a form
                        cy.get('.form').submit()

                    

                        
// Check element is visible
                        cy.get('.name').should('be.visible')

                        
// Check number of elements
                        cy.get('li').should('have.length', 4)

                        
// Check element has class
                        cy.get('.tree-item')
                            .should('not.have.class', 'active')
